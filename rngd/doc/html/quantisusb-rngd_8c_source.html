211 static void send_entropy(void)
212 {
213  static unsigned char send_buf[MAX_FRAME_SIZE];
214  int sock;
215  uint32_t write_size;
216  uint32_t header_size;
217  ssize_t send_status;
218  uint32_t send_len;
219  int clients_checked = 0;
220  uint32_t entropy_send;
221 
222  while(clients_checked < num_client_sockets) {
223  clients_checked++;
224 
225  if (!clients[receiver_index].keepalive_pending && !data_buf_available(data_buf)) {
226  goto next_receiver;
227  }
228 
229  /* Incomplete header */
230  if (clients[receiver_index].header_bytes_pending) {
231  header_size = clients[receiver_index].header_bytes_pending;
232  write_size = clients[receiver_index].entropy_pending;
233 
234  /* Incomplete entropy frame */
235  } else if (clients[receiver_index].entropy_pending) {
236  header_size = 0;
237  write_size = clients[receiver_index].entropy_pending;
238  } else {
239  header_size = HEADER_SIZE;
240  write_size = clients[receiver_index].entropy_requested;
241 
242  /* No entropy or keep-alive requested. Nothing to send. */
243  if (!write_size && !clients[receiver_index].keepalive_pending) {
244  goto next_receiver;
245  }
246  }
247 
248  sock = clients[receiver_index].socket;
249 
250  /* We have data to send but the socket is not available for write yet */
251  if (!FD_ISSET(sock, &writefds)) {
252  goto next_receiver;
253  }
254 
255  /* Never exceed max frame size */
256  if (write_size + header_size > MAX_FRAME_SIZE) {
257  write_size = MAX_FRAME_SIZE - header_size;
258  }
259 
260  if (write_size > data_buf_available(data_buf)) {
261  write_size = (uint32_t)data_buf_available(data_buf);
262  }
263 
264  /* Get random data from buffer */
265  write_size = (uint32_t)data_buf_read(data_buf, send_buf+header_size, write_size);
266 
267  /* Write the header */
268  if (header_size) {
269  if (!clients[receiver_index].header_bytes_pending) {
270  send_len = htonl(write_size);
271  memcpy(send_buf, &send_len, HEADER_SIZE);
272  } else {
273  /* Incomplete header */
274  unsigned char buf[HEADER_SIZE];
275  size_t buf_offset;
276 
277  /* Entropy pending cannot change without sending something
278  * so it is safe to assume that the bytes sent are still valid.
279  */
280  send_len = htonl(clients[receiver_index].entropy_pending);
281  memcpy(buf, &send_len, HEADER_SIZE);
282 
283  buf_offset = HEADER_SIZE - header_size;
284 
285  memcpy(send_buf, buf + buf_offset, header_size);
286  }
287  }
288 
289  /* Write entropy */
290  send_status = send(sock, send_buf, write_size+header_size, MSG_NOSIGNAL);
291  if (send_status >= 0) {
292  syslog(LOG_MAKEPRI(LOG_DAEMON, LOG_DEBUG), "Sent %d bytes of entropy to client", write_size);
293 
294  clients[receiver_index].keepalive_pending = 0;
295  clients[receiver_index].header_bytes_pending = 0;
296 
297  /* Incomplete header */
298  if (send_status < header_size) {
299  entropy_send = 0;
300  clients[receiver_index].header_bytes_pending = (uint32_t)(header_size-send_status);
301 
302  if (clients[receiver_index].entropy_pending) {
303  /* Already sending previous packet. Nothing to do here */
304  } else if (!write_size) {
305  /* No entropy requested. Mark keep-alive for later. */
306  clients[receiver_index].keepalive_pending = 1;
307  } else {
308  /* Mark the payload as pending */
309  clients[receiver_index].entropy_requested -= write_size;
310  clients[receiver_index].entropy_pending = write_size;
311  }
312 
313  } else if (header_size == HEADER_SIZE) {
314  entropy_send = (uint32_t)send_status - header_size;
315  clients[receiver_index].entropy_requested -= write_size;
316  clients[receiver_index].entropy_pending = write_size - entropy_send;
317  } else {
318  entropy_send = (uint32_t)send_status - header_size;
319  clients[receiver_index].entropy_pending -= entropy_send;
320  }
321 
322  /* Return unsent entropy to the data buffer */
323  if (entropy_send < write_size) {
324  data_buf_unread(data_buf, send_buf+send_status, write_size - entropy_send);
325  }
326 
327  } else {
328  /* Return unsent entropy to the buffer */
329  data_buf_unread(data_buf, send_buf+HEADER_SIZE, write_size);
330 
331  if (errno != EAGAIN && errno != EWOULDBLOCK && errno != EINTR) {
332  syslog(LOG_MAKEPRI(LOG_DAEMON, LOG_WARNING), "Send error: %s", strerror(errno));
333  }
334  }
335 
336 next_receiver:
337  receiver_advance();
338  }
339 }

420  if (opts < 0) {

427  syslog(LOG_MAKEPRI(LOG_DAEMON, LOG_ERR), "Error while setting socket to non-blocking. fcntl(F_SETFL) failed: %s", strerror(errno));
