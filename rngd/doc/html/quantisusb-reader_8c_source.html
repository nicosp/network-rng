95  fd_set writefds;

98  int select_status;
99  struct timeval timeout;

102  int benchmark = 1;
103 
104  signal(SIGTERM, onsigterm);

106 
107  ctx = quantis_usb_init(on_read, on_error, on_device, should_open_device, NULL, NULL);
108 
109  quantis_usb_enable_hotplug(ctx, 1);
110 
111  if (quantis_usb_device_count(ctx) < 1) {
112  fprintf(stderr, "No Quantis USB devices found\n");
113  quantis_usb_destroy(ctx);
114  return 1;
115  }
116 
117  if (benchmark) {
118  if (clock_gettime(CLOCK_MONOTONIC, &start)) {
119  fprintf(stderr, "Monotonic clock not available. Benchmark disabled\n");
120  benchmark = 0;
121  }
122  }
123 
124  while(!should_exit) {
125  FD_ZERO(&readfds);
126  FD_ZERO(&writefds);
127  FD_ZERO(&errorfds);
128 
129  timeout.tv_sec = 120;
130  timeout.tv_usec = 0;
131  nfds = 0;
132 
133  quantis_usb_read_all(ctx);
134 
135  select_status = quantis_usb_before_poll(ctx, &nfds, &readfds, &writefds, &errorfds, &timeout);
136  if (select_status) {
137  perror("Quantis error");
138  break;
139  }
140 
141  select_status = select(nfds, &readfds, &writefds, &errorfds, &timeout);
142  if (select_status < 0 && errno != EINTR) {
143  perror("select error");
144  break;
145  }
146 
147  select_status = quantis_usb_after_poll(ctx, !select_status, &readfds, &writefds, &errorfds);
148  if (select_status < 0) {
149  perror("Quantis error");
150  break;
151  }
152 
153  if (benchmark && total_bytes > BENCHMARK_BYTES) {
154  break;
155  }
156 
157  }
158 
159  if (benchmark) {
160  if (clock_gettime(CLOCK_MONOTONIC, &end)) {
161  fprintf(stderr, "Monotonic clock not available. Benchmark disabled\n");
162  } else {
163  time_t millis;
164 
165  millis = ((end.tv_sec - start.tv_sec) * 1000) + ((end.tv_nsec - start.tv_nsec) / 1000000);
166 
167  double seconds = (double)millis / 1000.0;
168 
169  fprintf(stderr, "Total read: %" PRIu64 ". Read rate: %.0f bytes/sec\n", total_bytes, (double)total_bytes / seconds);
170  }
171  }
172 
173  quantis_usb_destroy(ctx);
